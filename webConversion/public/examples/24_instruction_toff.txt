// TOFF - Timer Off-Delay Instruction
// Output stays ON for a delay after input turns OFF
// T0: 5 seconds delay = 50 * 0.1s
// T1: 3 seconds delay = 30 * 0.1s

// Press I0.0 - Q0.0 turns ON immediately
// Release I0.0 - Q0.0 stays ON for 5 more seconds
LD I0.0
TOFF T0, 50
LD T0
ST Q0.0

// Press I0.1 - Q0.1 turns ON immediately
// Release I0.1 - Q0.1 stays ON for 3 more seconds
LD I0.1
TOFF T1, 30
LD T1
ST Q0.1
